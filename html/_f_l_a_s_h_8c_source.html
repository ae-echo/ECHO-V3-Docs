11uint32_t Flash_GetSector(uint32_t Address)

16 if ((Address >= FLASH_BASE) && (Address < FLASH_BASE + FLASH_BANK_SIZE))
17 {
18 sector = (Address & ~FLASH_BASE) / FLASH_SECTOR_SIZE;

79{

86

93 // 16바이트 버퍼 초기화
94 for (j = 0; j < 16 && (i + j) < size; j++)

97 }
98
99 // 남은 공간은 0xFF로 패딩
100 while (j < 16)
101 {
102 DataBuffer[j++] = 0xFF;
103 }
104
105 // Flash에 데이터 기록 (128비트 단위)
106 if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_QUADWORD, flashAddress + i, (uint32_t *)DataBuffer) != HAL_OK)
107 {
108 HAL_FLASH_Lock();

110 }
111 }
