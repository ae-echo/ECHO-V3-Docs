168HAL_StatusTypeDef HAL_FLASHEx_Erase(FLASH_EraseInitTypeDef *pEraseInit, uint32_t *SectorError)
169{
170 HAL_StatusTypeDef status;
171 uint32_t sector_index;
172 __IO uint32_t *reg_cr;
173
174 /* Check the parameters */
175 assert_param(IS_FLASH_TYPEERASE(pEraseInit->TypeErase));
176
177 /* Process Locked */
178 __HAL_LOCK(&pFlash);
179
180 /* Reset error code */
181 pFlash.ErrorCode = HAL_FLASH_ERROR_NONE;
182
183 /* Wait for last operation to be completed */
184 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
185
186 if (status == HAL_OK)
187 {
188 /* Current operation type */
189 pFlash.ProcedureOnGoing = pEraseInit->TypeErase;
190
191 /* Access to SECCR or NSCR depends on operation type */
192#if defined (FLASH_OPTSR2_TZEN)
193 reg_cr = IS_FLASH_SECURE_OPERATION() ? &(FLASH->SECCR) : &(FLASH_NS->NSCR);
194#else
195 reg_cr = &(FLASH_NS->NSCR);
196#endif /* FLASH_OPTSR2_TZEN */
197
198 if ((pEraseInit->TypeErase & (~FLASH_NON_SECURE_MASK)) == FLASH_TYPEERASE_MASSERASE)
199 {
200 /* Mass erase to be done */
201 FLASH_MassErase(pEraseInit->Banks);
202
203 /* Wait for last operation to be completed */
204 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
205 }
206#if defined (FLASH_SR_OBKERR)
207 else if (pEraseInit->TypeErase == FLASH_TYPEERASE_OBK_ALT)
208 {
209 /* OBK erase to be done */
210 FLASH_OBKErase();
211
212 /* Wait for last operation to be completed */
213 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
214 }
215#endif /* FLASH_SR_OBKERR */
216 else
217 {
218 /* Initialization of SectorError variable */
219 *SectorError = 0xFFFFFFFFU;
220
221 /* Erase by sector by sector to be done*/
222 for (sector_index = pEraseInit->Sector; sector_index < (pEraseInit->NbSectors + pEraseInit->Sector); \
223 sector_index++)
224 {
225 FLASH_Erase_Sector(sector_index, pEraseInit->Banks);
226
227 /* Wait for last operation to be completed */
228 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
229
230 if (status != HAL_OK)
231 {
232 /* In case of error, stop erase procedure and return the faulty sector */
233 *SectorError = sector_index;
234 break;
235 }
236 }
237 }
238
239 /* If the erase operation is completed, disable the associated bits */
240 CLEAR_BIT((*reg_cr), (pEraseInit->TypeErase) & (~(FLASH_NON_SECURE_MASK)));
241 }
242
243 /* Process Unlocked */
244 __HAL_UNLOCK(&pFlash);
245
246 return status;
247}

256HAL_StatusTypeDef HAL_FLASHEx_Erase_IT(FLASH_EraseInitTypeDef *pEraseInit)

275 /* Process Unlocked */
276 __HAL_UNLOCK(&pFlash);

294 FLASH_IT_STRBERR | FLASH_IT_INCERR | FLASH_IT_OBKERR | \
295 FLASH_IT_OBKWERR);
296#else

298 FLASH_IT_STRBERR | FLASH_IT_INCERR);
299#endif /* FLASH_SR_OBKERR */
300
301 if ((pEraseInit->TypeErase & (~FLASH_NON_SECURE_MASK)) == FLASH_TYPEERASE_MASSERASE)
302 {
303 /* Mass erase to be done */
304 FLASH_MassErase(pEraseInit->Banks);

311 }
312#endif /* FLASH_SR_OBKERR */
313 else
314 {
315 /* Erase by sector to be done */
316 pFlash.NbSectorsToErase = pEraseInit->NbSectors;
317 pFlash.Sector = pEraseInit->Sector;
318
319 /* Erase first sector and wait for IT */
320 FLASH_Erase_Sector(pEraseInit->Sector, pEraseInit->Banks);
321 }
322 }
323
324 return status;

404 }

418

1867void HAL_FLASHEx_GetEccInfo(FLASH_EccInfoTypeDef *pData)
1868{
1869 uint32_t correction_reg = FLASH->ECCCORR;
1870 uint32_t detection_reg = FLASH->ECCDETR;
1871 uint32_t data_reg = FLASH->ECCDR;
1872 uint32_t addr_reg = 0xFFFFFFFFU;
1873
1874 /* Check if the operation is a correction or a detection*/
1875 if ((correction_reg & FLASH_ECCR_ECCC) != 0U)
1876 {
1877 /* Get area and offset address values from ECCCORR register*/
1878 pData->Area = correction_reg & (~(FLASH_ECCR_ECCIE | FLASH_ECCR_ADDR_ECC | FLASH_ECCR_ECCC));
1879 addr_reg = (correction_reg & FLASH_ECCR_ADDR_ECC);
1880 }
1881 else if ((detection_reg & FLASH_ECCR_ECCD) != 0U)
1882 {
1883 /* Get area and offset address values from ECCDETR register */
1884 pData->Area = detection_reg & (~(FLASH_ECCR_ADDR_ECC | FLASH_ECCR_ECCD));
1885 addr_reg = (detection_reg & FLASH_ECCR_ADDR_ECC);
1886 }
1887 else
1888 {
1889 /* Do nothing */
1890 }
1891
1892 /* Check that an ECC single or double error has occurred to continue the calculation of area address */
1893 if (addr_reg != 0xFFFFFFFFU)
1894 {
1895 /* Get address value according to area value*/
1896 switch (pData->Area)
1897 {
1898 case FLASH_ECC_AREA_USER_BANK1:
1899 /*
1900 * One error detection/correction or two error detections per 128-bit flash word
1901 * Therefore, the address returned by ECC registers in bank1 represents 128-bit flash word,
1902 * to get the correct address value, we must do a shift by 4 bits
1903 */
1904 addr_reg = addr_reg << 4U;
1905 pData->Address = FLASH_BASE + addr_reg;
1906 break;
1907 case FLASH_ECC_AREA_USER_BANK2:
1908 /*
1909 * One error detection/correction or two error detections per 128-bit flash word
1910 * Therefore, the address returned by ECC registers in bank2 represents 128-bit flash word,
1911 * to get the correct address value, we must do a shift by 4 bits
1912 */
1913 addr_reg = addr_reg << 4U;
1914 pData->Address = FLASH_BASE + FLASH_BANK_SIZE + addr_reg;
1915 break;
1916 case FLASH_ECC_AREA_SYSTEM:
1917 /* check system flash bank */
1918 if ((correction_reg & FLASH_ECCR_BK_ECC) == FLASH_ECCR_BK_ECC)
1919 {
1920 pData->Address = FLASH_SYSTEM_BASE + FLASH_SYSTEM_SIZE + addr_reg;
1921 }
1922 else
1923 {
1924 pData->Address = FLASH_SYSTEM_BASE + addr_reg;
1925 }
1926 break;
1927#if defined (FLASH_SR_OBKERR)
1928 case FLASH_ECC_AREA_OBK:
1929 pData->Address = FLASH_OBK_BASE + addr_reg;
1930 break;
1931#endif /* FLASH_SR_OBKERR */
1932#if defined (FLASH_EDATAR_EDATA_EN)
1933 case FLASH_ECC_AREA_EDATA:
1934 /* check flash high-cycle data bank */
1935 if ((correction_reg & FLASH_ECCR_BK_ECC) == FLASH_ECCR_BK_ECC)
1936 {
1937 /*
1938 * addr_reg is the address returned by the ECC register along with an offset value depends on area
1939 * To calculate the exact address set by user while an ECC occurred, we must subtract the offset value,
1940 * In addition, the address returned by ECC registers represents 128-bit flash word (multiply by 4),
1941 */
1942 pData->Address = FLASH_EDATA_BASE + FLASH_BANK_SIZE + ((addr_reg - FLASH_ADDRESS_OFFSET_EDATA) * 4U);
1943 }
1944 else
1945 {
1946 pData->Address = FLASH_EDATA_BASE + ((addr_reg - FLASH_ADDRESS_OFFSET_EDATA) * 4U);
1947 }
1948 break;
1949#endif /* FLASH_EDATAR_EDATA_EN */
1950 case FLASH_ECC_AREA_OTP:
1951 /* Address returned by the ECC is an halfword, multiply by 4 to get the exact address*/
1952 pData->Address = FLASH_OTP_BASE + ((addr_reg - FLASH_ADDRESS_OFFSET_OTP) * 4U);
1953 break;
1954
1955 default:
1956 /* Do nothing */
1957 break;
1958 }
1959 }
1960
1961 pData->Data = data_reg & FLASH_ECCR_ADDR_ECC;
1962}
