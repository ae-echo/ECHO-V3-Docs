166HAL_StatusTypeDef HAL_RS485Ex_Init(UART_HandleTypeDef *huart, uint32_t Polarity, uint32_t AssertionTime,
167 uint32_t DeassertionTime)
168{
169 uint32_t temp;
170
171 /* Check the UART handle allocation */
172 if (huart == NULL)
173 {
174 return HAL_ERROR;
175 }
176 /* Check the Driver Enable UART instance */
177 assert_param(IS_UART_DRIVER_ENABLE_INSTANCE(huart->Instance));
178
179 /* Check the Driver Enable polarity */
180 assert_param(IS_UART_DE_POLARITY(Polarity));
181
182 /* Check the Driver Enable assertion time */
183 assert_param(IS_UART_ASSERTIONTIME(AssertionTime));
184
185 /* Check the Driver Enable deassertion time */
186 assert_param(IS_UART_DEASSERTIONTIME(DeassertionTime));
187
188 if (huart->gState == HAL_UART_STATE_RESET)
189 {
190 /* Allocate lock resource and initialize it */
191 huart->Lock = HAL_UNLOCKED;
192
193#if (USE_HAL_UART_REGISTER_CALLBACKS == 1)
194 UART_InitCallbacksToDefault(huart);
195
196 if (huart->MspInitCallback == NULL)
197 {
198 huart->MspInitCallback = HAL_UART_MspInit;
199 }
200
201 /* Init the low level hardware */
202 huart->MspInitCallback(huart);
203#else
204 /* Init the low level hardware : GPIO, CLOCK, CORTEX */
205 HAL_UART_MspInit(huart);
206#endif /* (USE_HAL_UART_REGISTER_CALLBACKS) */
207 }
208
209 huart->gState = HAL_UART_STATE_BUSY;
210
211 /* Disable the Peripheral */
212 __HAL_UART_DISABLE(huart);
213
214 /* Perform advanced settings configuration */
215 /* For some items, configuration requires to be done prior TE and RE bits are set */
216 if (huart->AdvancedInit.AdvFeatureInit != UART_ADVFEATURE_NO_INIT)
217 {
218 UART_AdvFeatureConfig(huart);
219 }
220
221 /* Set the UART Communication parameters */
222 if (UART_SetConfig(huart) == HAL_ERROR)
223 {
224 return HAL_ERROR;
225 }
226
227 /* Enable the Driver Enable mode by setting the DEM bit in the CR3 register */
228 SET_BIT(huart->Instance->CR3, USART_CR3_DEM);
229
230 /* Set the Driver Enable polarity */
231 MODIFY_REG(huart->Instance->CR3, USART_CR3_DEP, Polarity);
232
233 /* Set the Driver Enable assertion and deassertion times */
234 temp = (AssertionTime << UART_CR1_DEAT_ADDRESS_LSB_POS);
235 temp |= (DeassertionTime << UART_CR1_DEDT_ADDRESS_LSB_POS);
236 MODIFY_REG(huart->Instance->CR1, (USART_CR1_DEDT | USART_CR1_DEAT), temp);
237
238 /* Enable the Peripheral */
239 __HAL_UART_ENABLE(huart);
240
241 /* TEACK and/or REACK to check before moving huart->gState and huart->RxState to Ready */
242 return (UART_CheckIdleState(huart));
243}

427HAL_StatusTypeDef HAL_UARTEx_StopModeWakeUpSourceConfig(UART_HandleTypeDef *huart, UART_WakeUpTypeDef WakeUpSelection)
428{
429 HAL_StatusTypeDef status = HAL_OK;
430 uint32_t tickstart;
431
432 /* check the wake-up from stop mode UART instance */
433 assert_param(IS_UART_WAKEUP_FROMSTOP_INSTANCE(huart->Instance));
434 /* check the wake-up selection parameter */
435 assert_param(IS_UART_WAKEUP_SELECTION(WakeUpSelection.WakeUpEvent));
436
437 /* Process Locked */
438 __HAL_LOCK(huart);
439
440 huart->gState = HAL_UART_STATE_BUSY;
441
442 /* Disable the Peripheral */
443 __HAL_UART_DISABLE(huart);
444
445 /* Set the wake-up selection scheme */
446 MODIFY_REG(huart->Instance->CR3, USART_CR3_WUS, WakeUpSelection.WakeUpEvent);
447
448 if (WakeUpSelection.WakeUpEvent == UART_WAKEUP_ON_ADDRESS)
449 {
450 UARTEx_Wakeup_AddressConfig(huart, WakeUpSelection);
451 }
452
453 /* Enable the Peripheral */
454 __HAL_UART_ENABLE(huart);
455
456 /* Init tickstart for timeout management */
457 tickstart = HAL_GetTick();
458
459 /* Wait until REACK flag is set */
460 if (UART_WaitOnFlagUntilTimeout(huart, USART_ISR_REACK, RESET, tickstart, HAL_UART_TIMEOUT_VALUE) != HAL_OK)
461 {
462 status = HAL_TIMEOUT;
463 }
464 else
465 {
466 /* Initialize the UART State */
467 huart->gState = HAL_UART_STATE_READY;
468 }
469
470 /* Process Unlocked */
471 __HAL_UNLOCK(huart);
472
473 return status;
474}

713HAL_StatusTypeDef HAL_UARTEx_ReceiveToIdle(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size, uint16_t *RxLen,
714 uint32_t Timeout)
715{
716 uint8_t *pdata8bits;
717 uint16_t *pdata16bits;
718 uint16_t uhMask;
719 uint32_t tickstart;
720
721 /* Check that a Rx process is not already ongoing */
722 if (huart->RxState == HAL_UART_STATE_READY)
723 {
724 if ((pData == NULL) || (Size == 0U))
725 {
726 return HAL_ERROR;
727 }
728
729#if defined(USART_DMAREQUESTS_SW_WA)
730 /* Disable the UART DMA Rx request if enabled */
731 if (HAL_IS_BIT_SET(huart->Instance->CR3, USART_CR3_DMAR))
732 {
733 CLEAR_BIT(huart->Instance->CR3, USART_CR3_DMAR);
734 }
735
736#endif /* USART_DMAREQUESTS_SW_WA */
737 huart->ErrorCode = HAL_UART_ERROR_NONE;
738 huart->RxState = HAL_UART_STATE_BUSY_RX;
739 huart->ReceptionType = HAL_UART_RECEPTION_TOIDLE;
740 huart->RxEventType = HAL_UART_RXEVENT_TC;
741
742 /* Init tickstart for timeout management */
743 tickstart = HAL_GetTick();
744
745 huart->RxXferSize = Size;
746 huart->RxXferCount = Size;
747
748 /* Computation of UART mask to apply to RDR register */
749 UART_MASK_COMPUTATION(huart);
750 uhMask = huart->Mask;
751
752 /* In case of 9bits/No Parity transfer, pRxData needs to be handled as a uint16_t pointer */
753 if ((huart->Init.WordLength == UART_WORDLENGTH_9B) && (huart->Init.Parity == UART_PARITY_NONE))
754 {
755 pdata8bits = NULL;
756 pdata16bits = (uint16_t *) pData;
757 }
758 else
759 {
760 pdata8bits = pData;
761 pdata16bits = NULL;
762 }
763
764 /* Initialize output number of received elements */
765 *RxLen = 0U;
766
767 /* as long as data have to be received */
768 while (huart->RxXferCount > 0U)
769 {
770 /* Check if IDLE flag is set */
771 if (__HAL_UART_GET_FLAG(huart, UART_FLAG_IDLE))
772 {
773 /* Clear IDLE flag in ISR */
774 __HAL_UART_CLEAR_FLAG(huart, UART_CLEAR_IDLEF);
775
776 /* If Set, but no data ever received, clear flag without exiting loop */
777 /* If Set, and data has already been received, this means Idle Event is valid : End reception */
778 if (*RxLen > 0U)
779 {
780 huart->RxEventType = HAL_UART_RXEVENT_IDLE;
781 huart->RxState = HAL_UART_STATE_READY;
782
783 return HAL_OK;
784 }
785 }
786
787 /* Check if RXNE flag is set */
788 if (__HAL_UART_GET_FLAG(huart, UART_FLAG_RXNE))
789 {
790 if (pdata8bits == NULL)
791 {
792 *pdata16bits = (uint16_t)(huart->Instance->RDR & uhMask);
793 pdata16bits++;
794 }
795 else
796 {
797 *pdata8bits = (uint8_t)(huart->Instance->RDR & (uint8_t)uhMask);
798 pdata8bits++;
799 }
800 /* Increment number of received elements */
801 *RxLen += 1U;
802 huart->RxXferCount--;
803 }
804
805 /* Check for the Timeout */
806 if (Timeout != HAL_MAX_DELAY)
807 {
808 if (((HAL_GetTick() - tickstart) > Timeout) || (Timeout == 0U))
809 {
810 huart->RxState = HAL_UART_STATE_READY;
811
812 return HAL_TIMEOUT;
813 }
814 }
815 }
816
817 /* Set number of received elements in output parameter : RxLen */
818 *RxLen = huart->RxXferSize - huart->RxXferCount;
819 /* At end of Rx process, restore huart->RxState to Ready */
820 huart->RxState = HAL_UART_STATE_READY;
821
822 return HAL_OK;
823 }
824 else
825 {
826 return HAL_BUSY;
827 }
828}

1021static void UARTEx_SetNbDataToProcess(UART_HandleTypeDef *huart)
1022{
1023 uint8_t rx_fifo_depth;
1024 uint8_t tx_fifo_depth;
1025 uint8_t rx_fifo_threshold;
1026 uint8_t tx_fifo_threshold;
1027 static const uint8_t numerator[] = {1U, 1U, 1U, 3U, 7U, 1U, 0U, 0U};
1028 static const uint8_t denominator[] = {8U, 4U, 2U, 4U, 8U, 1U, 1U, 1U};
1029
1030 if (huart->FifoMode == UART_FIFOMODE_DISABLE)
1031 {
1032 huart->NbTxDataToProcess = 1U;
1033 huart->NbRxDataToProcess = 1U;
1034 }
1035 else
1036 {
1037 rx_fifo_depth = RX_FIFO_DEPTH;
1038 tx_fifo_depth = TX_FIFO_DEPTH;
1039 rx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_RXFTCFG) >> USART_CR3_RXFTCFG_Pos);
1040 tx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_TXFTCFG) >> USART_CR3_TXFTCFG_Pos);
1041 huart->NbTxDataToProcess = ((uint16_t)tx_fifo_depth * numerator[tx_fifo_threshold]) /
1042 (uint16_t)denominator[tx_fifo_threshold];
1043 huart->NbRxDataToProcess = ((uint16_t)rx_fifo_depth * numerator[rx_fifo_threshold]) /
1044 (uint16_t)denominator[rx_fifo_threshold];
1045 }
1046}
