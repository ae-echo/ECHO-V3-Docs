393uint32_t HAL_SYSTICK_Config(uint32_t TicksNumb)
394{
395 if ((TicksNumb - 1UL) > SysTick_LOAD_RELOAD_Msk)
396 {
397 /* Reload value impossible */
398 return (1UL);
399 }
400
401 /* Set reload register */
402 WRITE_REG(SysTick->LOAD, (uint32_t)(TicksNumb - 1UL));
403
404 /* Load the SysTick Counter Value */
405 WRITE_REG(SysTick->VAL, 0UL);
406
407 /* Enable SysTick IRQ and SysTick Timer */
408 SET_BIT(SysTick->CTRL, (SysTick_CTRL_TICKINT_Msk | SysTick_CTRL_ENABLE_Msk));
409
410 /* Function successful */
411 return (0UL);
412}

424void HAL_SYSTICK_CLKSourceConfig(uint32_t CLKSource)
425{
426 /* Check the parameters */
427 assert_param(IS_SYSTICK_CLK_SOURCE(CLKSource));
428 switch (CLKSource)
429 {
430 /* Select HCLK as Systick clock source */
431 case SYSTICK_CLKSOURCE_HCLK:
432 SET_BIT(SysTick->CTRL, SYSTICK_CLKSOURCE_HCLK);
433 break;
434 /* Select HCLK_DIV8 as Systick clock source */
435 case SYSTICK_CLKSOURCE_HCLK_DIV8:
436 CLEAR_BIT(SysTick->CTRL, SYSTICK_CLKSOURCE_HCLK);
437 MODIFY_REG(RCC->CCIPR4, RCC_CCIPR4_SYSTICKSEL, (0x00000000U));
438 break;
439 /* Select LSI as Systick clock source */
440 case SYSTICK_CLKSOURCE_LSI:
441 CLEAR_BIT(SysTick->CTRL, SYSTICK_CLKSOURCE_HCLK);
442 MODIFY_REG(RCC->CCIPR4, RCC_CCIPR4_SYSTICKSEL, RCC_CCIPR4_SYSTICKSEL_0);
443 break;
444 /* Select LSE as Systick clock source */
445 case SYSTICK_CLKSOURCE_LSE:
446 CLEAR_BIT(SysTick->CTRL, SYSTICK_CLKSOURCE_HCLK);
447 MODIFY_REG(RCC->CCIPR4, RCC_CCIPR4_SYSTICKSEL, RCC_CCIPR4_SYSTICKSEL_1);
448 break;
449 default:
450 /* Nothing to do */
451 break;
452 }
453}

463uint32_t HAL_SYSTICK_GetCLKSourceConfig(void)
464{
465 uint32_t systick_source;
466 uint32_t systick_rcc_source;
467
468 /* Read SysTick->CTRL register for internal or external clock source */
469 if (READ_BIT(SysTick->CTRL, SysTick_CTRL_CLKSOURCE_Msk) != 0U)
470 {
471 /* Internal clock source */
472 systick_source = SYSTICK_CLKSOURCE_HCLK;
473 }
474 else
475 {
476 /* External clock source, check the selected one in RCC */
477 systick_rcc_source = READ_BIT(RCC->CCIPR4, RCC_CCIPR4_SYSTICKSEL);
478
479 switch (systick_rcc_source)
480 {
481 case (0x00000000U):
482 systick_source = SYSTICK_CLKSOURCE_HCLK_DIV8;
483 break;
484
485 case (RCC_CCIPR4_SYSTICKSEL_0):
486 systick_source = SYSTICK_CLKSOURCE_LSI;
487 break;
488
489 case (RCC_CCIPR4_SYSTICKSEL_1):
490 systick_source = SYSTICK_CLKSOURCE_LSE;
491 break;
492
493 default:
494 systick_source = SYSTICK_CLKSOURCE_HCLK_DIV8;
495 break;
496 }
497 }
498 return systick_source;
499}

551void HAL_MPU_Enable(uint32_t MPU_Control)
552{
553 __DMB(); /* Data Memory Barrier operation to force any outstanding writes to memory before enabling the MPU */
554
555 /* Enable the MPU */
556 MPU->CTRL = MPU_Control | MPU_CTRL_ENABLE_Msk;
557
558 /* Enable fault exceptions */
559 SCB->SHCSR |= SCB_SHCSR_MEMFAULTENA_Msk;
560
561 /* Follow ARM recommendation with */
562 /* Data Synchronization and Instruction Synchronization Barriers to ensure MPU configuration */
563 __DSB(); /* Ensure that the subsequent instruction is executed only after the write to memory */
564 __ISB(); /* Flush and refill pipeline with updated MPU configuration settings */
565}

783static void MPU_ConfigRegion(MPU_Type *MPUx, const MPU_Region_InitTypeDef *const pMPU_RegionInit)
784{
785 /* Check the parameters */
786#if defined (__ARM_FEATURE_CMSE) && (__ARM_FEATURE_CMSE == 3U)
787 assert_param(IS_MPU_INSTANCE(MPUx));
788#endif /* __ARM_FEATURE_CMSE */
789 assert_param(IS_MPU_REGION_NUMBER(pMPU_RegionInit->Number));
790 assert_param(IS_MPU_REGION_ENABLE(pMPU_RegionInit->Enable));
791 assert_param(IS_MPU_INSTRUCTION_ACCESS(pMPU_RegionInit->DisableExec));
792 assert_param(IS_MPU_REGION_PERMISSION_ATTRIBUTE(pMPU_RegionInit->AccessPermission));
793 assert_param(IS_MPU_ACCESS_SHAREABLE(pMPU_RegionInit->IsShareable));
794
795 /* Follow ARM recommendation with Data Memory Barrier prior to MPU configuration */
796 __DMB();
797
798 /* Set the Region number */
799 MPUx->RNR = pMPU_RegionInit->Number;
800
801 /* Disable the Region */
802 CLEAR_BIT(MPUx->RLAR, MPU_RLAR_EN_Msk);
803
804 MPUx->RBAR = (((uint32_t)pMPU_RegionInit->BaseAddress & 0xFFFFFFE0UL) |
805 ((uint32_t)pMPU_RegionInit->IsShareable << MPU_RBAR_SH_Pos) |
806 ((uint32_t)pMPU_RegionInit->AccessPermission << MPU_RBAR_AP_Pos) |
807 ((uint32_t)pMPU_RegionInit->DisableExec << MPU_RBAR_XN_Pos));
808
809 MPUx->RLAR = (((uint32_t)pMPU_RegionInit->LimitAddress & 0xFFFFFFE0UL) |
810 ((uint32_t)pMPU_RegionInit->AttributesIndex << MPU_RLAR_AttrIndx_Pos) |
811 ((uint32_t)pMPU_RegionInit->Enable << MPU_RLAR_EN_Pos));
812}

824static void MPU_ConfigMemoryAttributes(MPU_Type *MPUx, const MPU_Attributes_InitTypeDef *const pMPU_AttributesInit)
825{
826 __IO uint32_t *p_mair;
827 uint32_t attr_values;
828 uint32_t attr_number;
829
830 /* Check the parameters */
831#if defined (__ARM_FEATURE_CMSE) && (__ARM_FEATURE_CMSE == 3U)
832 assert_param(IS_MPU_INSTANCE(MPUx));
833#endif /* __ARM_FEATURE_CMSE */
834 assert_param(IS_MPU_ATTRIBUTES_NUMBER(pMPU_AttributesInit->Number));
835 /* No need to check Attributes value as all 0x0..0xFF possible */
836
837 /* Follow ARM recommendation with Data Memory Barrier prior to MPUx configuration */
838 __DMB();
839
840 if (pMPU_AttributesInit->Number < MPU_ATTRIBUTES_NUMBER4)
841 {
842 /* Program MPU_MAIR0 */
843 p_mair = &(MPUx->MAIR0);
844 attr_number = pMPU_AttributesInit->Number;
845 }
846 else
847 {
848 /* Program MPU_MAIR1 */
849 p_mair = &(MPUx->MAIR1);
850 attr_number = (uint32_t)pMPU_AttributesInit->Number - 4U;
851 }
852
853 attr_values = *(p_mair);
854 attr_values &= ~(0xFFUL << (attr_number * 8U));
855 *(p_mair) = attr_values | ((uint32_t)pMPU_AttributesInit->Attributes << (attr_number * 8U));
856}
