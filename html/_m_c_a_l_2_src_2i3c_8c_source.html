114uint32_t I3C_ENTDAA(uint8_t dynAddr, I3C_DeviceType device)
115{
116 uint8_t retryCount = 1;
117 uint8_t isMatched = 0;
118 uint32_t i3c_error;
119 const uint8_t* refPayload = NULL; // 비교할 기준 페이로드 포인터 (const 적용)
120
121 // ENUM 값에 따라 비교 대상 선택
122 switch (device)
123 {
124 case I3C_DEVICE_DW9785:
125 refPayload = DW9785_payload;
126 break;
127 case I3C_DEVICE_DW97XX:
128 refPayload = DW97xx_payload;
129 break;
130 default:
131 return HAL_I3C_ERROR_INVALID_PARAM; // 잘못된 입력
132 }
133
134 dynAddr >>= 1; // Dynamic Address 변환
135
136 do
137 {
138 targetPayload = 0;
139 i3c_status = HAL_I3C_Ctrl_DynAddrAssign(&hi3c1, &targetPayload, I3C_RSTDAA_THEN_ENTDAA, 100);
140 isMatched = isPayloadMatched(refPayload);
141
142 if (i3c_status == HAL_BUSY)
143 {
144 HAL_I3C_Ctrl_SetDynAddr(&hi3c1, isMatched ? dynAddr : dynAddr + retryCount);
145 if (!isMatched) retryCount++;
146 }
147
148 } while (i3c_status == HAL_BUSY);
149
150 i3c_error = hi3c1.ErrorCode;
151 return i3c_error;
152}

160uint8_t isPayloadMatched(const uint8_t* refPayload)
161{
162 for (int i = 0; i < 8; i++)
163 {
164 uint8_t extractedByte = (targetPayload >> (i * 8)) & 0xFF; // targetPayload에서 바이트 추출
165
166 if (extractedByte != refPayload[i])

168 return 0; // 불일치
169 }
170 }
171 return 1; // 일치

182{

185

191 };
192
193 /* Direct CCC 프레임 설정 */
194 I3C_CCCTypeDef directCCC[] =
