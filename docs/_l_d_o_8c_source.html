96void LOD_SetVoltage(LDO_Select LDO, float Voltage)
97{
98 uint8_t step = LDO_GetStep(Voltage);
99 if(g_LDO_Selects[LDO] > step)
100 {
101 LDO_controldown(LDO, g_LDO_Selects[LDO] - step);
102 g_LDO_Selects[LDO] = g_LDO_Selects[LDO] - (g_LDO_Selects[LDO] - step);

107 g_LDO_Selects[LDO] = g_LDO_Selects[LDO] + (step - g_LDO_Selects[LDO]);
108 }
109 else
110 {}

119uint8_t LDO_GetStep(float Inputvoltage)
120{
121 uint8_t closestStep = 0;
122 float minDifference = fabs(Inputvoltage - VoltageTable[0]);
123
124 for (uint8_t step = 1; step < STEPMAX; step++) {
125 float difference = fabs(Inputvoltage - VoltageTable[step]);
126 if (difference < minDifference) {
127 minDifference = difference;
128 closestStep = step;
129 }
130 }
131
132 return closestStep;
133}

140float LDO_GetVoltage(uint8_t Step)
141{
142 int R17 = 3;
143 int R16 = 200;
144 int R21 = 10;
145 float MCP4011Offset = 0.78125;
146 float MCP4011 = Step * MCP4011Offset;
147 float R_s = MCP4011 + R17;
148 float R_s2 = R16 * R_s /(R16 + R_s);
149 float Voltage = (R21 +R_s2) / R21 *0.8;
150
151 return Voltage;
152}
