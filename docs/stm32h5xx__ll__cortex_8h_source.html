510__STATIC_INLINE void LL_MPU_Enable(uint32_t MPU_Control)
511{
512 __DMB(); /* Data Memory Barrier operation to force any outstanding writes to memory before enabling the MPU */
513
514 /* Enable the MPU*/
515 MPU->CTRL = MPU_CTRL_ENABLE_Msk | MPU_Control;
516
517 /* Follow ARM recommendation with */
518 /* Data Synchronization and Instruction Synchronization Barriers to ensure MPU configuration */
519 __DSB(); /* Ensure that the subsequent instruction is executed only after the write to memory */
520 __ISB(); /* Flush and refill pipeline with updated MPU configuration settings */
521}
