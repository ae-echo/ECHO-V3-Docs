201void SystemInit(void)
202{
203 uint32_t reg_opsr;
204
205 /* FPU settings ------------------------------------------------------------*/
206 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
207 SCB->CPACR |= ((3UL << 20U)|(3UL << 22U)); /* set CP10 and CP11 Full Access */
208 #endif
209
210 /* Reset the RCC clock configuration to the default reset state ------------*/
211 /* Set HSION bit */
212 RCC->CR = RCC_CR_HSION;
213
214 /* Reset CFGR register */
215 RCC->CFGR1 = 0U;
216 RCC->CFGR2 = 0U;
217
218 /* Reset HSEON, HSECSSON, HSEBYP, HSEEXT, HSIDIV, HSIKERON, CSION, CSIKERON, HSI48 and PLLxON bits */
219#if defined(RCC_CR_PLL3ON)
220 RCC->CR &= ~(RCC_CR_HSEON | RCC_CR_HSECSSON | RCC_CR_HSEBYP | RCC_CR_HSEEXT | RCC_CR_HSIDIV | RCC_CR_HSIKERON | \
221 RCC_CR_CSION | RCC_CR_CSIKERON |RCC_CR_HSI48ON | RCC_CR_PLL1ON | RCC_CR_PLL2ON | RCC_CR_PLL3ON);
222#else
223 RCC->CR &= ~(RCC_CR_HSEON | RCC_CR_HSECSSON | RCC_CR_HSEBYP | RCC_CR_HSEEXT | RCC_CR_HSIDIV | RCC_CR_HSIKERON | \
224 RCC_CR_CSION | RCC_CR_CSIKERON |RCC_CR_HSI48ON | RCC_CR_PLL1ON | RCC_CR_PLL2ON);
225#endif
226
227 /* Reset PLLxCFGR register */
228 RCC->PLL1CFGR = 0U;
229 RCC->PLL2CFGR = 0U;
230#if defined(RCC_CR_PLL3ON)
231 RCC->PLL3CFGR = 0U;
232#endif /* RCC_CR_PLL3ON */
233
234 /* Reset PLL1DIVR register */
235 RCC->PLL1DIVR = 0x01010280U;
236 /* Reset PLL1FRACR register */
237 RCC->PLL1FRACR = 0x00000000U;
238 /* Reset PLL2DIVR register */
239 RCC->PLL2DIVR = 0x01010280U;
240 /* Reset PLL2FRACR register */
241 RCC->PLL2FRACR = 0x00000000U;
242#if defined(RCC_CR_PLL3ON)
243 /* Reset PLL3DIVR register */
244 RCC->PLL3DIVR = 0x01010280U;
245 /* Reset PLL3FRACR register */
246 RCC->PLL3FRACR = 0x00000000U;
247#endif /* RCC_CR_PLL3ON */
248
249 /* Reset HSEBYP bit */
250 RCC->CR &= ~(RCC_CR_HSEBYP);
251
252 /* Disable all interrupts */
253 RCC->CIER = 0U;
254
255 /* Configure the Vector Table location add offset address ------------------*/
256 #ifdef VECT_TAB_SRAM
257 SCB->VTOR = SRAM1_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
258 #else
259 SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
260 #endif /* VECT_TAB_SRAM */
261
262 /* Check OPSR register to verify if there is an ongoing swap or option bytes update interrupted by a reset */
263 reg_opsr = FLASH->OPSR & FLASH_OPSR_CODE_OP;
264 if ((reg_opsr == FLASH_OPSR_CODE_OP) || (reg_opsr == (FLASH_OPSR_CODE_OP_2 | FLASH_OPSR_CODE_OP_1)))
265 {
266 /* Check FLASH Option Control Register access */
267 if ((FLASH->OPTCR & FLASH_OPTCR_OPTLOCK) != 0U)
268 {
269 /* Authorizes the Option Byte registers programming */
270 FLASH->OPTKEYR = 0x08192A3BU;
271 FLASH->OPTKEYR = 0x4C5D6E7FU;
272 }
273 /* Launch the option bytes change operation */
274 FLASH->OPTCR |= FLASH_OPTCR_OPTSTART;
275
276 /* Lock the FLASH Option Control Register access */
277 FLASH->OPTCR |= FLASH_OPTCR_OPTLOCK;
278 }
279}

323void SystemCoreClockUpdate(void)
324{
325 uint32_t pllp, pllsource, pllm, pllfracen, hsivalue, tmp;
326 float_t fracn1, pllvco;
327
328 /* Get SYSCLK source -------------------------------------------------------*/
329 switch (RCC->CFGR1 & RCC_CFGR1_SWS)
330 {
331 case 0x00UL: /* HSI used as system clock source */
332 SystemCoreClock = (uint32_t) (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3));
333 break;
334
335 case 0x08UL: /* CSI used as system clock source */
336 SystemCoreClock = CSI_VALUE;
337 break;
338
339 case 0x10UL: /* HSE used as system clock source */
340 SystemCoreClock = HSE_VALUE;
341 break;
342
343 case 0x18UL: /* PLL1 used as system clock source */
344 /* PLL_VCO = (HSE_VALUE or HSI_VALUE or CSI_VALUE/ PLLM) * PLLN
345 SYSCLK = PLL_VCO / PLLR
346 */
347 pllsource = (RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1SRC);
348 pllm = ((RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1M)>> RCC_PLL1CFGR_PLL1M_Pos);
349 pllfracen = ((RCC->PLL1CFGR & RCC_PLL1CFGR_PLL1FRACEN)>>RCC_PLL1CFGR_PLL1FRACEN_Pos);
350 fracn1 = (float_t)(uint32_t)(pllfracen* ((RCC->PLL1FRACR & RCC_PLL1FRACR_PLL1FRACN)>> RCC_PLL1FRACR_PLL1FRACN_Pos));
351
352 switch (pllsource)
353 {
354 case 0x01UL: /* HSI used as PLL clock source */
355 hsivalue = (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3)) ;
356 pllvco = ((float_t)hsivalue / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1N) + \
357 (fracn1/(float_t)0x2000) +(float_t)1 );
358 break;
359
360 case 0x02UL: /* CSI used as PLL clock source */
361 pllvco = ((float_t)CSI_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1N) + \
362 (fracn1/(float_t)0x2000) +(float_t)1 );
363 break;
364
365 case 0x03UL: /* HSE used as PLL clock source */
366 pllvco = ((float_t)HSE_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1N) + \
367 (fracn1/(float_t)0x2000) +(float_t)1 );
368 break;
369
370 default: /* No clock sent to PLL*/
371 pllvco = (float_t) 0U;
372 break;
373 }
374
375 pllp = (((RCC->PLL1DIVR & RCC_PLL1DIVR_PLL1P) >>RCC_PLL1DIVR_PLL1P_Pos) + 1U ) ;
376 SystemCoreClock = (uint32_t)(float_t)(pllvco/(float_t)pllp);
377
378 break;
379
380 default:
381 SystemCoreClock = HSI_VALUE;
382 break;
383 }
384 /* Compute HCLK clock frequency --------------------------------------------*/
385 /* Get HCLK prescaler */
386 tmp = AHBPrescTable[((RCC->CFGR2 & RCC_CFGR2_HPRE) >> RCC_CFGR2_HPRE_Pos)];
387 /* HCLK clock frequency */
388 SystemCoreClock >>= tmp;
389}
