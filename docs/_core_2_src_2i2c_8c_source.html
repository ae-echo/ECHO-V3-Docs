40uint32_t I2C_GPIO_Write_Func(uint8_t Addr, uint8_t* pData, uint16_t size, uint32_t tmout, int pDataOffset)
41{
42 uint32_t i, tmp;
43 uint32_t ack;
44
45 // Start
46 SDA(0);
47 delay_i2c(tmout); // 100ns
48 ack = 0;
49 tmp = Addr;
50 SCL(0);
51
52 // Start Addr
53 for(i=0; i<8; i++)
54 {
55 I2C_Port.pPort->BSRR = (Addr & 0x80) ? I2C_Port.SDA : (I2C_Port.SDA << 16); // Check point
56
57 SDA((tmp & 0x80) ? 1 : 0);
58 delay_i2c(tmout);
59 SCL(1);
60 delay_i2c(tmout);
61 SCL(0);
62 tmp <<= 1;
63 }
64
65 // ACK Event
66 delay_i2c(tmout); /*hold margin*/
67 SDA(1); /*Set High-z*/
68 delay_i2c(tmout);
69 SCL(1);
70 SCL(1);
71 delay_i2c(tmout);
72 ack = (I2C_Port.pPort->IDR & I2C_Port.SDA) ? 1 : 0;
73 SCL(0);
74 delay_i2c(tmout);
75
76 while(size > 0)
77 {
78 //tmp = (*pData++);
79 tmp = *pData;
80 pData = pData + pDataOffset;
81
82 for(i=0; i<8; i++)
83 {
84 SDA((tmp & 0x80) ? 1 : 0);
85 delay_i2c(tmout);
86 SCL(1);
87 delay_i2c(tmout);
88 SCL(0);
89 tmp <<= 1;
90 }
91 size--;
92
93 //ACK Event
94 delay_i2c(tmout); /*hold margin*/
95 SDA(1);
96 delay_i2c(tmout);
97 SCL(1);
98 delay_i2c(tmout);
99 SCL(0);
100 }
101
102 // Stop
103 SDA(0);
104 delay_i2c(tmout);
105 SCL(1);
106 delay_i2c(tmout);
107 SDA(1);
108
109 return ack;
110}
