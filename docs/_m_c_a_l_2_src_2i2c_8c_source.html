80void I2C_ClearSlaveStatus(void)
81{
82 if(I2C_RxFlag == I2C_SLAVE_RECV)
83 {
84 memset(Slave_I2C_Buf.RxBuf, 0, sizeof(Slave_I2C_Buf.RxBuf));
85 I2C_RxFlag = I2C_SLAVE_READY;
86 }
87 else if(I2C_RxFlag == I2C_SLAVE_RECV_SEQ) //Sequence
88 {
89 bool Cmd_Check = false;
90
91 while(!Cmd_Check)
92 {
93 if(Slave_I2C_Buf.Rxcnt <= 2) //데이터 모두 처리
94 {
95 memset(Slave_I2C_Buf.RxBuf, 0, sizeof(Slave_I2C_Buf.RxBuf));
96 I2C_RxFlag = I2C_SLAVE_READY;
97 return;
98 }
99
100 Slave_I2C_Buf.RxBuf[0]++; //Reg Addr 주소 증가
101 remove_index(Slave_I2C_Buf.RxBuf, Slave_I2C_Buf.Rxcnt , 1);
102 Slave_I2C_Buf.Rxcnt--; //Reg Data 남은수
103 Cmd_Check = is_valid_command(Slave_I2C_Buf.RxBuf[0]);
104 }
105 }
106}

417void HAL_I2C_ListenCpltCallback(I2C_HandleTypeDef *hi2c)
418{
419 if (hi2c->Instance == I2C2)
420 {
421 int recvCnt = REG_SIZE - hi2c->XferSize;
422
423 if (recvCnt == 1) // 마스터측 Read Reg
424 {
425 // Read 요청을 위한 Register Address 설정 단계
426 Slave_I2C_Buf.RegAddr = Slave_I2C_Buf.RxBuf[0];
427 Slave_I2C_Buf.Direction = I2C_SLAVE_SEND;
428 }
429 else // 마스터측 전송 완료
430 {
431 if(Slave_I2C_Buf.Direction == I2C_SLAVE_SEND) //Master Read
432 {
433 }
434 else //Master Send
435 {
436 Slave_I2C_Buf.Rxcnt = recvCnt;
437
438 uint8_t* dest = &I2C_RegBuf[Slave_I2C_Buf.RxBuf[0]];
439 uint8_t* src = &Slave_I2C_Buf.RxBuf[1];
440 uint8_t length = Slave_I2C_Buf.Rxcnt - 1; //주소제외
441
442 memcpy(dest, src, length);
443
444 I2C_RxFlag = Slave_I2C_Buf.Rxcnt > 2 ? I2C_SLAVE_RECV_SEQ : I2C_SLAVE_RECV;
445 }
446
447 Slave_I2C_Buf.Direction = I2C_SLAVE_READY;
448 }

450 // Read 완료 후 Listen 모드 재진입
451 HAL_I2C_EnableListen_IT(hi2c);

456void HAL_I2C_AddrCallback(I2C_HandleTypeDef *hi2c, uint8_t TransferDirection, uint16_t AddrMatchCode)
457{
458
459 if (hi2c->Instance == I2C2)
460 {
461 if (AddrMatchCode != hi2c->Init.OwnAddress1 || I2C_RxFlag == I2C_SLAVE_RECV || I2C_RxFlag == I2C_SLAVE_RECV_SEQ)
462 {
463 // 잘못된 주소 요청 → NACK 전송
464 hi2c->Instance->CR2 |= I2C_CR2_NACK;
465 return;
466 }
467
468 // 기존 처리 로직...
469 if (TransferDirection == I2C_DIRECTION_TRANSMIT) // Master가 Write
470 {
471 HAL_I2C_Slave_Seq_Receive_IT(hi2c, Slave_I2C_Buf.RxBuf, sizeof(Slave_I2C_Buf.RxBuf), I2C_NEXT_FRAME);
472 }
473 else if (TransferDirection == I2C_DIRECTION_RECEIVE) // Master가 Read
474 {
475 HAL_I2C_Slave_Seq_Transmit_IT(hi2c, &Slave_I2C_Buf.TxBuf[Slave_I2C_Buf.RegAddr], REG_SIZE, I2C_NEXT_FRAME);
476 }
477 }
