719__STATIC_INLINE void LL_GPIO_LockPin(GPIO_TypeDef *GPIOx, uint32_t PinMask)
720{
721 __IO uint32_t temp;
722 WRITE_REG(GPIOx->LCKR, GPIO_LCKR_LCKK | PinMask);
723 WRITE_REG(GPIOx->LCKR, PinMask);
724 WRITE_REG(GPIOx->LCKR, GPIO_LCKR_LCKK | PinMask);
725 /* Read LCKK register. This read is mandatory to complete key lock sequence */
726 temp = READ_REG(GPIOx->LCKR);
727 (void) temp;
